import re

def clean_string(s: str) -> str:
    """Remove all non-alphabetical characters from string."""
    return ''.join(re.findall(r'[A-Za-z]', s))


def is_mirror(str1: str, str2: str) -> bool:
    # Clean both strings (remove non-alphabetical chars)
    cleaned1 = clean_string(str1)
    cleaned2 = clean_string(str2)

    # Compare cleaned1 with reverse of cleaned2
    return cleaned1 == cleaned2[::-1]
